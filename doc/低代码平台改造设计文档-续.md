class LLMNodeBuilder(NodeBuilder):
    """LLM节点构建器"""
    
    def build(self, config: Dict[str, Any]) -> Callable:
        async def llm_node(state: AgentState) -> AgentState:
            from app.llm.client import build_llm_client
            
            provider_id = config.get('provider', 'openai')
            model = config.get('model', 'gpt-4')
            temperature = config.get('temperature', 0.7)
            prompt_template = config.get('prompt_template', '')
            
            client = build_llm_client(provider_id=provider_id, model=model)
            
            messages = state['messages'].copy()
            if prompt_template:
                messages.insert(0, {
                    "role": "system",
                    "content": prompt_template
                })
            
            response = await client.chat(
                messages=messages,
                temperature=temperature
            )
            
            state['messages'].append(response)
            state['response'] = response.get('content', '')
            
            return state
        
        return llm_node

class KnowledgeNodeBuilder(NodeBuilder):
    """知识检索节点构建器"""
    
    def build(self, config: Dict[str, Any]) -> Callable:
        async def knowledge_node(state: AgentState) -> AgentState:
            from app.rag.retriever import RAGRetriever
            
            kb_id = config.get('knowledge_base_id')
            top_k = config.get('top_k', 5)
            
            # 获取用户查询
            query = state['messages'][-1]['content']
            
            # 检索知识
            retriever = RAGRetriever(kb_id)
            docs = await retriever.retrieve(query, top_k=top_k)
            
            # 将检索结果注入状态
            state['retrieved_docs'] = docs
            
            # 构建上下文
            context = "\n\n".join([doc['content'] for doc in docs])
            state['context'] = {'knowledge': context}
            
            return state
        
        return knowledge_node

class ToolExecutorNodeBuilder(NodeBuilder):
    """工具执行节点构建器"""
    
    def build(self, config: Dict[str, Any]) -> Callable:
        async def tool_executor_node(state: AgentState) -> AgentState:
            tools = config.get('tools', [])
            max_iterations = config.get('max_iterations', 5)
            
            # 调用现有的 executor_node
            from app.graph.nodes.executor import executor_node
            
            # 注入工具配置
            state['available_tools'] = tools
            
            # 执行工具调用
            result_state = await executor_node(state)
            
            return result_state
        
        return tool_executor_node

        workflow_json=request.workflow_json,

        created_by=user_id
    )

        "id": workflow.id,
        "version": workflow.version,
        "created_at": workflow.created_at.isoformat()
    }

@router.post("/{workflow_id}/stream")
async def stream_workflow(
    workflow_id: int,
    request: WorkflowExecuteRequest
):
    """流式执行工作流（SSE）"""
    from app.store.workflow_store import WorkflowStore
    from app.workflow.parser import WorkflowParser
    from app.graph.state import AgentState
    import asyncio
    
    async def event_stream():
        try:
            # 加载工作流
            store = WorkflowStore()
            workflow = await store.get(workflow_id)
            
            if not workflow:
                yield f"data: {json.dumps({'error': '工作流不存在'})}\n\n"
                return
            
            # 解析并构建LangGraph
            parser = WorkflowParser()
            graph = parser.parse(json.loads(workflow.workflow_json))
            
            # 初始化状态
            initial_state: AgentState = {
                "messages": [{"role": "user", "content": request.message}],
                "session_id": request.session_id,
                # ... 其他字段初始化
            }
            
            # 流式执行
            async for event in graph.astream_events(initial_state):
                # 发送节点开始事件
                if event['event'] == 'on_chain_start':
                    yield f"data: {json.dumps({'event': 'node_start', 'node_id': event['name']})}\n\n"
                
                # 发送节点完成事件
                elif event['event'] == 'on_chain_end':
                    yield f"data: {json.dumps({'event': 'node_complete', 'node_id': event['name'], 'result': event['data']})}\n\n"
            
            # 发送完成事件
            yield f"data: {json.dumps({'event': 'workflow_complete'})}\n\n"
        
        except Exception as e:
            yield f"data: {json.dumps({'event': 'error', 'message': str(e)})}\n\n"
    
    return StreamingResponse(
        event_stream(),
        media_type="text/event-stream"
    )

// web/src/components/WorkflowEditor/index.tsx
import React, { useCallback, useState } from 'react';
import ReactFlow, {
  Node,
  Edge,
  addEdge,
  Connection,
  useNodesState,
  useEdgesState,
  Controls,
  MiniMap,
  Background,
  Panel,
} from 'reactflow';
import 'reactflow/dist/style.css';

import { NodeLibrary } from './NodeLibrary';
import { ConfigPanel } from './ConfigPanel';
import { nodeTypes } from './nodes';

interface WorkflowEditorProps {
  initialNodes?: Node[];
  initialEdges?: Edge[];
  onSave: (workflow: { nodes: Node[]; edges: Edge[] }) => void;
}

export function WorkflowEditor({ initialNodes = [], initialEdges = [], onSave }: WorkflowEditorProps) {
  const [nodes, setNodes, onNodesChange] = useNodesState(initialNodes);
  const [edges, setEdges, onEdgesChange] = useEdgesState(initialEdges);
  const [selectedNode, setSelectedNode] = useState<Node | null>(null);

  const onConnect = useCallback((params: Connection) => {
    setEdges((eds) => addEdge({ 
      ...params, 
      animated: true,
      type: 'smoothstep',
    }, eds));
  }, []);

  const onNodeClick = useCallback((_: React.MouseEvent, node: Node) => {
    setSelectedNode(node);
  }, []);

  const onNodeDragStop = useCallback((_: React.MouseEvent, node: Node) => {
    // 自动保存位置
    console.log('Node moved:', node.id, node.position);
  }, []);

  const handleSave = useCallback(() => {
    onSave({ nodes, edges });
  }, [nodes, edges, onSave]);

  const handleExportJSON = useCallback(() => {
    const workflowJson = {
      nodes: nodes.map(n => ({
        id: n.id,
        type: n.type,
        config: n.data.config || {},
        position: n.position,
      })),
      edges: edges.map(e => ({
        id: e.id,
        from: e.source,
        to: e.target,
        condition: e.label,
      })),
    };
    
    const blob = new Blob([JSON.stringify(workflowJson, null, 2)], { type: 'application/json' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = 'workflow.json';
    a.click();
    URL.revokeObjectURL(url);
  }, [nodes, edges]);

  return (
    <div className="workflow-editor" style={{ display: 'flex', height: '100%' }}>
      {/* 左侧节点库 */}
      <div style={{ width: 200, borderRight: '1px solid #e8e8e8', padding: 16 }}>
        <NodeLibrary />
      </div>

      {/* 中间画布 */}
      <div style={{ flex: 1, position: 'relative' }}>
        <ReactFlow
          nodes={nodes}
          edges={edges}
          onNodesChange={onNodesChange}
          onEdgesChange={onEdgesChange}
          onConnect={onConnect}
          onNodeClick={onNodeClick}
          onNodeDragStop={onNodeDragStop}
          nodeTypes={nodeTypes}
          fitView
        >
          <Controls />
          <MiniMap />
          <Background variant="dots" gap={12} size={1} />
          
          <Panel position="top-right">
            <button onClick={handleSave} style={{ marginRight: 8 }}>
              保存
            </button>
            <button onClick={handleExportJSON}>
              导出JSON
            </button>
          </Panel>
        </ReactFlow>
      </div>

      {/* 右侧配置面板 */}
      {selectedNode && (
        <div style={{ width: 300, borderLeft: '1px solid #e8e8e8', padding: 16 }}>
          <ConfigPanel 
            node={selectedNode} 
            onUpdate={(config) => {
              setNodes((nds) =>
                nds.map((n) =>
                  n.id === selectedNode.id
                    ? { ...n, data: { ...n.data, config } }
                    : n
                )
              );
            }}
          />
        </div>
      )}
    </div>
  );
}

    initial_state: AgentState = {
        "messages": [{"role": "user", "content": "测试消息"}],
        # ... 初始化其他字段
    }
    
    final_state = await graph.ainvoke(initial_state)
